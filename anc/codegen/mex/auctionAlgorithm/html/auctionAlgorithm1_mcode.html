1function [Customer2Item,Item2Customer]=auctionAlgorithm(RewardMatrix)
 2
 3epsilon=0.05; %amount of deviation from the optimal reward
 4
 5[NofCustomers,NofItems]=size(RewardMatrix);
 6if (NofCustomers>NofItems)
 7    error('Number of columns must be greater than or equal to the number of rows');
 8end
 9
10Item2Customer=zeros(1,NofItems);
11Customer2Item=zeros(1,NofCustomers);
12
13while ~isempty(find(Customer2Item==0, 1))
14    if (NofItems==1) %if there is only one item
15        [~, Item2Customer]=max(RewardMatrix);%Assign the item to the best customer
16        Customer2Item(Item2Customer)=1;%Assign the corresponding customer to the item
17    else
18        for i=1:NofCustomers
19            if ~Customer2Item(i)
20                [maxval,maxind]=max(RewardMatrix(i,:));%find maximum element value and its index
21                RewardMatrix(i,maxind)=min(RewardMatrix(i,:))-1;%make the maximum minimum to find second maximum
22                [secondmaxval,~]=max(RewardMatrix(i,:));%find the second maximum value and its index
23                RewardMatrix(i,maxind)=maxval; %restore the maximum value
24
25                Customer2Item(i)=maxind; %Assign the customer the item
26                if Item2Customer(maxind) %if item is already assigned
27                    Customer2Item(Item2Customer(maxind))=0;%unassign the corresponding customer
28                end
29                Item2Customer(maxind)=i; %Assign the item to the customer
30                RewardMatrix(:,maxind)=RewardMatrix(:,maxind)-(maxval-secondmaxval+epsilon);%reduce the item's value
31            end
32        end
33    end
34end
